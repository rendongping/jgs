### 3. 什么是 CRDT？与 OT 有什么区别？

**参考答案**：

CRDT 是具有数学特性的数据结构，允许多个副本独立修改并最终一致。OT 需要中央服务器对操作进行变换。

CRDT 更适合离线优先、去中心化场景；OT 适合强一致、中心化场景。

**评分维度**：

### 8. CRDT 的核心数据结构有哪些？分别适用于什么场景？

**参考答案**：

**1. GCounter (Grow-only Counter)**
```js
// 只能递增的计数器
class GCounter {
  constructor() {
    this.counters = new Map(); // nodeId -> value
  }
  
  increment(nodeId) {
    this.counters.set(nodeId, (this.counters.get(nodeId) || 0) + 1);
  }
  
  merge(other) {
    for (const [nodeId, value] of other.counters) {
      this.counters.set(nodeId, Math.max(this.counters.get(nodeId) || 0, value));
    }
  }
  
  value() {
    return [...this.counters.values()].reduce((a, b) => a + b, 0);
  }
}
```

**2. LWW-Register (Last-Writer-Wins Register)**
```js
class LWWRegister {
  constructor() {
    this.state = { value: null, timestamp: 0, nodeId: "" };
  }
  
  assign(value, timestamp, nodeId) {
    if (timestamp > this.state.timestamp || 
        (timestamp === this.state.timestamp && nodeId > this.state.nodeId)) {
      this.state = { value, timestamp, nodeId };
    }
  }
  
  merge(other) {
    this.assign(other.state.value, other.state.timestamp, other.state.nodeId);
  }
}
```

**3. OR-Set (Observed-Remove Set)**
```js
class ORSet {
  constructor() {
    this.elements = new Map(); // element -> Set<tag>
    this.tombstones = new Map();
  }
  
  add(element, tag) {
    if (!this.elements.has(element)) this.elements.set(element, new Set());
    this.elements.get(element).add(tag);
  }
  
  remove(element) {
    const tags = this.elements.get(element);
    if (tags) {
      this.tombstones.set(element, new Set(tags));
      this.elements.delete(element);
    }
  }
}
```

**4. RGA (Replicated Growable Array)** - 文本编辑的核心 CRDT，使用 linked list 实现有序列表，插入性能好。

### 9. OT 算法的核心原理是什么？如何处理并发冲突？

**参考答案**：

OT 通过操作变换（Operation Transformation）来保持文档一致性。当两个并发操作到达服务器时，服务器会对其中一个操作进行变换，使其适用于另一个操作应用后的文档状态。

```js
// OT 核心函数：transform(op1, op2) -> [op1', op2']
// 保证 apply(apply(doc, op1), op2') === apply(apply(doc, op2), op1')

class TextOperation {
  constructor() {
    this.ops = [];
  }
  
  retain(n) { this.ops.push({ type: "retain", n }); return this; }
  insert(s) { this.ops.push({ type: "insert", s }); return this; }
  delete(n) { this.ops.push({ type: "delete", n }); return this; }
  
  static transform(op1, op2) {
    const op1Prime = new TextOperation();
    const op2Prime = new TextOperation();
    let i = 0, j = 0;
    
    while (i < op1.ops.length && j < op2.ops.length) {
      const a = op1.ops[i];
      const b = op2.ops[j];
      
      if (a.type === "insert" && b.type === "insert") {
        op1Prime.insert(a.s);
        op2Prime.insert(b.s);
        i++; j++;
      } else if (a.type === "delete" && b.type === "delete") {
        const minLen = Math.min(a.n, b.n);
        op1Prime.delete(minLen);
        op2Prime.delete(minLen);
        a.n -= minLen; b.n -= minLen;
        if (a.n === 0) i++;
        if (b.n === 0) j++;
      }
      // ... 其他组合处理
    }
    return [op1Prime, op2Prime];
  }
  
  apply(doc) {
    let result = "";
    let pos = 0;
    for (const op of this.ops) {
      if (op.type === "retain") {
        result += doc.slice(pos, pos + op.n);
        pos += op.n;
      } else if (op.type === "insert") {
        result += op.s;
      } else if (op.type === "delete") {
        pos += op.n;
      }
    }
    result += doc.slice(pos);
    return result;
  }
}
```

**OT vs CRDT 选择**：
| 场景 | 推荐方案 | 原因 |
|------|---------|------|
| 中心化协作（Google Docs） | OT | 强一致，服务器可控制 |
| 离线优先 / P2P | CRDT | 无需中央协调 |
| 大规模实时编辑 | CRDT | 无中心瓶颈 |

**评分维度**：
- 变换原理理解（30%）
- 实现细节（25%）
- 冲突处理（25%）
- OT vs CRDT 选型（20%）

**常见错误**：
- 认为 OT 是单一算法，实际有多种变体（Jupiter、dOPT、COT）。
- 未处理操作变换的组合爆炸问题。
- 低估 OT 实现复杂度，尤其是 undo/redo。

```js
          }
        }
```
```js
      }
```

**参考答案**：

```js
// WebRTC 信令流程
// 1. 用户 A 创建 Offer -> 2. 通过信令服务器转发给 B
// 3. 用户 B 创建 Answer -> 4. 通过信令服务器转发给 A
// 5. 交换 ICE Candidate

class SignalingServer {
  constructor(wsServer) {
    this.wsServer = wsServer;
    this.rooms = new Map();
    
    wsServer.on("connection", (ws) => {
      ws.on("message", (msg) => this.handleMessage(ws, JSON.parse(msg)));
    });
  }
  
  handleMessage(ws, data) {
    switch (data.type) {
      case "join_room":
        this.joinRoom(ws, data.roomId, data.userId);
        break;
      case "offer":
        // 转发给房间内指定用户
        this.sendToUser(data.targetUserId, {
          type: "offer",
          sdp: data.sdp,
          from: ws.userId
        });
        break;
      case "answer":
        this.sendToUser(data.targetUserId, {
          type: "answer",
          sdp: data.sdp,
          from: ws.userId
        });
        break;
      case "ice_candidate":
        this.sendToUser(data.targetUserId, {
          type: "ice_candidate",
          candidate: data.candidate,
          from: ws.userId
        });
        break;
      case "leave_room":
        this.leaveRoom(ws, data.roomId);
        break;
    }
  }
  
  joinRoom(ws, roomId, userId) {
    ws.roomId = roomId;
    ws.userId = userId;
    
    if (!this.rooms.has(roomId)) {
      this.rooms.set(roomId, new Map());
    }
    this.rooms.get(roomId).set(userId, ws);
    
    // 通知房间其他用户
    this.broadcast(roomId, {
      type: "user_joined",
      userId
    }, [userId]);
  }
}
```

```js
// 连接恢复策略

// 1. 消息确认机制 (ACK)
class ReliableWebSocket {
  constructor(url) {
    this.url = url;
    this.ws = null;
    this.pendingMessages = new Map();
    this.messageId = 0;
    this.ackTimeout = 5000;
    this.maxRetries = 3;
  }
  
  connect() {
    this.ws = new WebSocket(this.url);
    
    this.ws.onopen = () => {
      if (this.lastMessageId) {
        this.send({ type: "recover", lastMessageId: this.lastMessageId });
      }
      this.resendPending();
    };
    
    this.ws.onmessage = (event) => {
      const msg = JSON.parse(event.data);
      if (msg.type === "ack") {
        this.handleAck(msg.messageId);
      } else if (msg.type === "recover_response") {
        msg.messages.forEach(m => this.handleMessage(m));
      }
    };
    
    this.ws.onclose = () => this.scheduleReconnect();
  }
  
  send(data) {
    const id = ++this.messageId;
    const message = { id, ...data };
    this.pendingMessages.set(id, { data: message, retries: 0 });
    this.ws.send(JSON.stringify(message));
    setTimeout(() => this.handleTimeout(id), this.ackTimeout);
  }
  
  handleAck(messageId) {
    this.pendingMessages.delete(messageId);
  }
  
  resendPending() {
    for (const [id, msg] of this.pendingMessages) {
      if (msg.retries < this.maxRetries) {
        msg.retries++;
        this.ws.send(JSON.stringify(msg.data));
      } else {
        this.onMessageFailed?.(msg.data);
        this.pendingMessages.delete(id);
      }
    }
  }
  
  scheduleReconnect() {
    const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempt || 0), 30000);
    this.reconnectAttempt = (this.reconnectAttempt || 0) + 1;
    setTimeout(() => this.connect(), delay);
  }
}
```

```js
// 向量时钟 (Vector Clock) 保证偏序
class VectorClock {
  constructor(nodeId) {
    this.nodeId = nodeId;
    this.clock = new Map();
  }
  
  tick() {
    this.clock.set(this.nodeId, (this.clock.get(this.nodeId) || 0) + 1);
  }
  
  static compare(a, b) {
    let aG = false, bG = false;
    const all = new Set([...a.clock.keys(), ...b.clock.keys()]);
    for (const n of all) {
      const av = a.clock.get(n) || 0;
      const bv = b.clock.get(n) || 0;
      if (av > bv) aG = true;
      if (bv > av) bG = true;
    }
    if (!aG && !bG) return "equal";
    if (aG && !bG) return "happen_before";
    if (!aG && bG) return "happen_after";
    return "concurrent";
  }
}

  isProcessed(opId) {
    return this.processedOps.has(opId);
  }
  
  markProcessed(opId) {
    this.processedOps.add(opId);
    setTimeout(() => this.processedOps.delete(opId), this.ttl);
  }
}
```

```js
// 服务端背压控制
class BackpressureManager {
  constructor() {
    this.maxBufferSize = 1000;
    this.maxPendingMessages = 100;
    this.clientBuffers = new Map();
  }
  
  canSend(clientId, messageSize) {
    const buffer = this.getBuffer(clientId);
    if (buffer.size + messageSize > this.maxBufferSize) return false;
    if (buffer.pendingCount >= this.maxPendingMessages) return false;
    return true;
  }
  
  sendWithBackpressure(clientId, message) {
    if (!this.canSend(clientId, JSON.stringify(message).length)) {
      this.sendToClient(clientId, { type: "flow_control", action: "pause" });
      return false;
    }
    this.bufferMessage(clientId, message);
    this.flush(clientId);
    return true;
  }
}

// 客户端自适应流控
class AdaptiveFlowController {
  constructor() {
    this.windowSize = 10;
    this.rtt = 0;
    this.pending = 0;
  }
  
  onMessageSent() { this.pending++; }
  
  onMessageAck(sendTime) {
    this.pending--;
    const rttMs = Date.now() - sendTime;
    this.rtt = this.rtt * 0.7 + rttMs * 0.3;
    if (++this.successCount % 5 === 0) {
      this.windowSize = Math.min(this.windowSize + 1, 100);
    }
  }
  
  onMessageTimeout() {
    this.pending--;
    this.windowSize = Math.max(Math.floor(this.windowSize / 2), 1);
  }
  
  canSend() { return this.pending < this.windowSize; }
}
```

```js
class LatencyTester {
  constructor(ws) {
    this.ws = ws;
    this.probes = [];
  }
  
  measureLatency() {
    const id = Date.now() + Math.random();
    const sendTime = performance.now();
    this.ws.send(JSON.stringify({ type: "ping", id }));
    this.ws.onmessage = (event) => {
```

```js
class SpatialIndex {
  constructor() {
    this.grid = new Map();
    this.tileSize = 100;
  }
  
  queryViewport(vp) {
    const result = new Set();
    for (let x = vp.x; x <= vp.x + vp.width; x += this.tileSize) {
      for (let y = vp.y; y <= vp.y + vp.height; y += this.tileSize) {
        const key = `${Math.floor(x / this.tileSize)},${Math.floor(y / this.tileSize)}`;
        if (this.grid.has(key)) {
          for (const id of this.grid.get(key)) result.add(id);
        }
      }
    }
    return [...result];
  }
}
```
